import sympy as sp
sp.init_printing(use_latex=True)

def L(f):
  return sp.laplace_transform(f,t,s, noconds=True)

def invL(F):
  return sp.inverse_laplace_transform(F,s,t)

t,s = sp.symbols('t,s')
a = sp.symbols('a', real=True, positive=True)

def r(t):
  return t * sp.Heaviside(t)
